📚 この設計仕様書について（初心者向け解説）

この設計仕様書は、要件定義書で決めた「何を作るか」を具体的に「どう作るか」に落とし込んだ技術文書です。

プログラマーや設計者が実際にシステムを構築する際の詳細な設計図になります。

🎯 設計仕様書の目的

  • 技術選定の理由を明確化
  • 実装方法を具体的に指示
  • 品質基準を設定
  • 開発チームの共通理解を促進

📐 設計の主要ポイント

  • シンプルな構造：保守しやすさを重視
  • モジュール化：機能ごとに独立した設計
  • レスポンシブ：あらゆるデバイスに対応
  • パフォーマンス：高速で軽量な動作

🔧 技術解説（初心者向け）

💻 「設計」って何をすること？

設計とは、建物を建てる前に設計図を描くのと同じように、プログラムを作る前に「どんな技術を使って」「どんな構造で作るか」を決めることです。

4.2 分電盤シミュレーター

⚠️ 複雑性注意: 分電盤シミュレーターは最も複雑な実装となるため、段階的な開発アプローチを採用
class DistributionBoardSimulator { constructor(container) { this.container = container; this.board = { mainBreaker: { capacity: 60, status: 'on', load: 0 }, branchBreakers: [ { id: 1, name: 'リビング', capacity: 20, status: 'on', load: 0 }, { id: 2, name: '寝室', capacity: 20, status: 'on', load: 0 }, { id: 3, name: 'キッチン', capacity: 30, status: 'on', load: 0 } ], earthLeakage: { sensitivity: 30, status: 'normal' } }; this.appliances = new Map(); this.init(); } calculateTotalLoad() { return this.board.branchBreakers .filter(breaker => breaker.status === 'on') .reduce((total, breaker) => total + breaker.load, 0); } checkOverload(breaker, newLoad) { if (newLoad > breaker.capacity) { this.tripBreaker(breaker); this.showAlert(`${breaker.name}のブレーカーが過負荷により遮断されました`); return true; } return false; } addAppliance(breakerId, appliance) { const breaker = this.board.branchBreakers.find(b => b.id === breakerId); const newLoad = breaker.load + appliance.power; if (!this.checkOverload(breaker, newLoad)) { breaker.load = newLoad; this.appliances.set(appliance.id, { breakerId, appliance }); this.updateDisplay(); } } }

4.3 電線許容電流値計算エンジン

class AllowableCurrentCalculator { constructor() { this.cableData = { 'vvf': { '1.6': { air: 19, conduit: 17, bundle: 15, buried: 16 }, '2.0': { air: 27, conduit: 23, bundle: 19, buried: 21 }, '2.6': { air: 36, conduit: 31, bundle: 26, buried: 28 }, // ... 他のサイズ }, 'iv': { /* IVケーブルデータ */ }, 'cv': { /* CVケーブルデータ */ }, 'cvt': { /* CVTケーブルデータ */ } }; } calculate(params) { const { cableType, size, installation, temperature, actualCurrent } = params; // 基準許容電流の取得 const baseCurrent = this.cableData[cableType][size][installation]; // 温度補正係数の計算 const tempCorrection = this.calculateTemperatureCorrection(temperature); // 許容電流の算出 const allowableCurrent = baseCurrent * tempCorrection; // 使用率の計算 const usagePercent = (actualCurrent / allowableCurrent) * 100; // 安全性評価 const safety = this.evaluateSafety(usagePercent); return { baseCurrent, tempCorrection, allowableCurrent, usagePercent, safety, steps: this.generateCalculationSteps(params, baseCurrent, tempCorrection, allowableCurrent, usagePercent) }; } calculateTemperatureCorrection(temperature) { const baseTemp = 30; // 基準温度 30℃ if (temperature > baseTemp) { // 高温時の補正（1℃につき1.5%減少） return 1.0 - (temperature - baseTemp) * 0.015; } else if (temperature < baseTemp) { // 低温時の補正（1℃につき1.0%増加） return 1.0 + (baseTemp - temperature) * 0.01; } return 1.0; } evaluateSafety(usagePercent) { if (usagePercent <= 70) return { level: 'safe', message: '安全範囲内' }; if (usagePercent <= 85) return { level: 'caution', message: '注意が必要' }; if (usagePercent <= 100) return { level: 'warning', message: '危険範囲' }; return { level: 'danger', message: '過負荷状態' }; } }

6.2 JavaScript最適化

// デバウンス関数による処理最適化 function debounce(func, wait) { let timeout; return function executedFunction(...args) { const later = () => { clearTimeout(timeout); func(...args); }; clearTimeout(timeout); timeout = setTimeout(later, wait); }; } // requestAnimationFrameによるアニメーション最適化 class AnimationManager { constructor() { this.animations = new Set(); this.isRunning = false; } add(animation) { this.animations.add(animation); if (!this.isRunning) { this.start(); } } remove(animation) { this.animations.delete(animation); if (this.animations.size === 0) { this.stop(); } } start() { this.isRunning = true; this.tick(); } tick() { if (this.isRunning) { this.animations.forEach(animation => animation.update()); requestAnimationFrame(() => this.tick()); } } stop() { this.isRunning = false; } }

6.3 メモリ管理

class ResourceManager { constructor() { this.eventListeners = new Map(); this.intervals = new Set(); this.observers = new Set(); } addEventListener(element, event, handler) { if (!this.eventListeners.has(element)) { this.eventListeners.set(element, new Map()); } this.eventListeners.get(element).set(event, handler); element.addEventListener(event, handler); } cleanup() { // イベントリスナーの削除 this.eventListeners.forEach((events, element) => { events.forEach((handler, event) => { element.removeEventListener(event, handler); }); }); // インターバルの削除 this.intervals.forEach(interval => clearInterval(interval)); // オブザーバーの削除 this.observers.forEach(observer => observer.disconnect()); // マップとセットをクリア this.eventListeners.clear(); this.intervals.clear(); this.observers.clear(); } }
